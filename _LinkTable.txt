    and Year(LinkTable.Fecha) <= Year(Today());

        Fecha                                                                 as SK.Fecha,

        Capitalize(Upper(Date(Fecha, 'MMMM'))) 		                          as Calendario.Mes,
        Year(Fecha) 		                                                  as Calendario.AÃ±o,
